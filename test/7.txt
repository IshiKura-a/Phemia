    printf("hello");
    if(low<high)
    	{
    		int key = a[low];
    		int i=low;
    		int j=high;
    		while(i<j)
    		{
    			while((a[j]>=key)&&(i<j)) {j--;}
    			a[i]=a[j];
    			while((a[i]<=key)&&(i<j)) {i++;}
    			a[j]=a[i];
    		}
    		a[i]=key;
    		qsort(a, low,i-1);
    		qsort(a, i+1,high);
    	}
};

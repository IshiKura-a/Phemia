[25][25]int x = new [25][25]int();
[25][25]int y = new [25][25]int();

int m1;
int n1;
int m2;
int n2;

int i;
int j;

int tmp;

scanf("%d%d", m1, n1);
for (i=0; i<m1; i++) {
    for (j=0; j<n1; j++) {
        scanf("%d", tmp);
        x[i][j] = tmp;
    }
}

scanf("%d%d", m2, n2);
for (i=0; i<m2; i++) {
    for (j=0; j<n2; j++) {
        scanf("%d", tmp);
        y[i][j] = tmp;
    }
}

int m3 = m1;
int n3 = n2;

if (n1 != m2) {
    printf("Incompatible Dimensions\n");
} else {
    int k;
    for (i=0; i<m3; i++) {
        for (j=0; j<n3; j++) {
            int sum = 0;
            for (k=0; k<n1; k++) {
                sum = sum + x[i][k] * y[k][j];
            }
            printf("%10d", sum);
        }
        printf("\n");
    }
}

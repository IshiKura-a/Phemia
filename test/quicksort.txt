int patition(int n[],int l,int r){
    int p=n[l];
    int i=l;
    int j=r;
    while(i<j){
        while(n[j]>=p&&i<j){
            j--;
        }
        while(n[i]<=p&&i<j){
            i++;
        }
        int temp=n[i];
        n[i]=n[j];
        n[j]=temp;
    }
    int temp1=n[l];
    n[l]=n[i];
    n[i]=temp1;
    return i;
}
int sort(int n[],int l,int r){
    if(l<r){
        int index=patition(n,l,r);
        sort(n,l,index-1);
        sort(n,index+1,r);
    }
    return 0;
}
int quicksort(int n[]){
    sort(n,0,n.length-1);
    return 0;
}
int main(){
    int n[]={7,5,6,3,2,8,1};
    quicksort(n);
    return 0;
}
